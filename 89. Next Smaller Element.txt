#include <bits/stdc++.h>
vector<int> nextSmallerElement(vector<int> &arr, int n)
{
    stack<int> stack;
    stack.push(-1);

    vector<int> ans(n);

    for(int i = n - 1; i >= 0; i--){
        if(stack.top() < arr[i]) {
            ans[i] = stack.top();
            stack.push(arr[i]);
        } else {
            while(stack.top() >= arr[i]) {
                stack.pop();
            }
            ans[i] = stack.top();
            stack.push(arr[i]);
        }
    }

    return ans;
}
